pipeline {
    agent any
    stages
    {
        stage('Test') {
            steps {
                script {
                    try {
                        bat 'mvn -X test'
                    } catch (Exception e) {
                        echo "Build failed: ${e.message}"
                        currentBuild.result = 'FAILURE'
                    }
                }
            }
        }
        stage('Test1')
        {
            steps {
                echo "Testing the Code.."
                bat "mvn test"
            }
        }
        stage('Compile')
        {
            steps {
                echo "Testing the project.."
                bat "mvn compile"
            }
        }
        
    }
}
