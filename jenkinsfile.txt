pipeline {
    agent any

    environment {
        MAVEN_HOME = MAVEN_HOME // Ensure you have a Maven tool configured in Jenkins
        PATH = "${env.MAVEN_HOME}/bin:${env.PATH}"
    }

    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/kpanj/ProjectLJShopSite.git'
            }
        }
        
        stage('Build') {
            steps {
                script {
                    // Clean and compile the project
                    sh 'mvn clean compile'
                }
            }
        }
        
        stage('Test') {
            steps {
                script {
                    // Run the tests
                    sh 'mvn test'
                }
            }
        }
        
        stage('Package') {
            steps {
                script {
                    // Package the application
                    sh 'mvn package'
                }
            }
        }
        
        stage('Deploy') {
            steps {
                script {
                    // Deploy the application
                    sh 'mvn deploy'
                }
            }
        }
    }

    post {
        always {
            // Archive the artifacts, e.g., JAR/WAR files
            archiveArtifacts artifacts: 'target/*.jar', allowEmptyArchive: true

            // Publish test results
            junit 'target/surefire-reports/*.xml'

            // Clean up workspace
            cleanWs()
        }
        success {
            echo 'Build was successful!'
        }
        failure {
            echo 'Build failed!'
        }
    }
}
